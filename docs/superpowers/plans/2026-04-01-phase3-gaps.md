"""
Script 27 — Full Indicator Screen (P0)
========================================
Phase 3 only tested 5 hand-coded indicators for bust prediction.
QEngine has 174 jesse-rust indicators available.

This script screens ALL of them:
  1. Compute every indicator on 20yr EUR-USD 5m data
  2. Rank by AUC for bust prediction (same method as script 21)
  3. Test top indicators as entry gates (same method as script 22)
  4. Compare against the original ER_100 winner

Questions:
  Is Efficiency Ratio really the best bust predictor?
  Are there stronger signals we missed?
  Do volatility indicators (BBW, Damiani, Squeeze) outperform chop indicators?

Outputs:
  - results/27_indicator_screen.json
  - plots/27_auc_ranking.png
  - plots/27_top_gates.png
  - plots/27_category_comparison.png
"""
from utils import *

print("=" * 60)
print("Script 27 — Full Indicator Screen (P0)")
print("=" * 60)

# ══════════════════════════════════════════════════════════════
# 1. LOAD DATA & BASELINE
# ══════════════════════════════════════════════════════════════
print("\n[1/5] Loading data and running baseline...")
candles = load_candles(timeframe='5m', start_date='2006-01-03', end_date='2025-12-30')
print(f"  Loaded {len(candles):,} candles")

cfg = SimConfig(sizing_curve='sqrt', sizing_factor=2.0, max_levels=12,
                hedge_dist_pips=10, tp_pips=20)
signals = random_signals(len(candles), avg_spacing=50, rng=np.random.default_rng(42))

# Run baseline
baseline_results = run_cycles_on_signals(candles, signals, cfg)
baseline = cycle_summary(baseline_results)
print(f"  Baseline: {baseline['n_cycles']} cycles, bust={baseline['bust_rate']:.2f}%, PF={baseline['profit_factor']:.2f}")

busts = [c for c in baseline_results if c.bust]
wins = [c for c in baseline_results if c.is_win]

# ══════════════════════════════════════════════════════════════
# 2. COMPUTE ALL INDICATORS
# ══════════════════════════════════════════════════════════════
print("\n[2/5] Computing all indicators (this may take several minutes)...")
all_features = compute_all_indicators(candles)

# Also include the original phase3 hand-coded features for comparison
phase3_features = compute_chop_features(candles, windows=(50, 100, 200))
for k, v in phase3_features.items():
    all_features[f'p3_{k}'] = v  # Prefix to distinguish

print(f"  Total features: {len(all_features)}")

# ══════════════════════════════════════════════════════════════
# 3. RANK BY BUST PREDICTION AUC
# ══════════════════════════════════════════════════════════════
print("\n[3/5] Ranking all indicators by bust prediction AUC...")

indicator_auc = {}
for k, vals in sorted(all_features.items()):
    bust_vals = []
    win_vals = []
    for c in busts:
        idx = c.entry_idx
        if idx < len(vals) and not np.isnan(vals[idx]):
            bust_vals.append(vals[idx])
    for c in wins:
        idx = c.entry_idx
        if idx < len(vals) and not np.isnan(vals[idx]):
            win_vals.append(vals[idx])
    
    if len(bust_vals) < 5 or len(win_vals) < 5:
        continue
    
    # Mann-Whitney U for AUC
    n_busts_v = len(bust_vals)
    n_wins_v = len(win_vals)
    pairs = sorted(zip(bust_vals + win_vals, [1]*n_busts_v + [0]*n_wins_v), key=lambda x: x[0])
    concordant = 0
    n_pos_seen = 0
    for val, label in pairs:
        if label == 0:
            concordant += n_pos_seen
        else:
            n_pos_seen += 1
    
    auc = concordant / (n_busts_v * n_wins_v) if n_busts_v * n_wins_v > 0 else 0.5
    auc = max(auc, 1 - auc)  # Ensure > 0.5
    
    # Cohen's d
    bm, wm = np.mean(bust_vals), np.mean(win_vals)
    bs, ws = np.std(bust_vals), np.std(win_vals)
    pooled = np.sqrt((bs**2 + ws**2) / 2) if (bs + ws) > 0 else 1
    cohens_d = abs(bm - wm) / pooled
    
    indicator_auc[k] = {
        'auc': round(auc, 4),
        'cohens_d': round(cohens_d, 4),
        'bust_mean': round(bm, 6),
        'win_mean': round(wm, 6),
        'bust_higher': bm > wm,
    }

# Rank
ranked = sorted(indicator_auc.items(), key=lambda x: x[1]['auc'], reverse=True)
print(f"\n  Top 20 bust predictors:")
for name, info in ranked[:20]:
    print(f"    {name:35s}: AUC={info['auc']:.4f}  d={info['cohens_d']:.3f}  "
          f"bust_{'>' if info['bust_higher'] else '<'}win")

# Was ER_100 really the best?
er100_rank = next((i for i, (n, _) in enumerate(ranked) if 'er_100' in n.lower() or 'efficiency_ratio_100' in n.lower()), -1)
print(f"\n  Original winner (ER_100) ranks #{er100_rank + 1} out of {len(ranked)}")

# Plot: top 30 by AUC
fig, ax = plt.subplots(figsize=(12, 10))
top_n = min(30, len(ranked))
names_r = [r[0] for r in ranked[:top_n]]
aucs = [r[1]['auc'] for r in ranked[:top_n]]
# Color by category
def categorize(name):
    vol_kw = ['atr', 'bbw', 'natr', 'damiani', 'cvi', 'ui', 'stddev', 'trange', 'mass', 'waddah', 'squeeze']
    trend_kw = ['adx', 'er_', 'pfe', 'chop', 'dx', 'aroon', 'supertrend', 'di_', 'dm_', 'stiffness', 'trendflex', 'cfo', 'fosc']
    mom_kw = ['rsi', 'cci', 'mfi', 'willr', 'mom', 'roc', 'trix', 'tsi', 'ao', 'bop', 'cmo', 'dti', 'rvgi', 'rsx', 'stoch']
    for kw in vol_kw:
        if kw in name.lower():
            return 'volatility'
    for kw in trend_kw:
        if kw in name.lower():
            return 'trend'
    for kw in mom_kw:
        if kw in name.lower():
            return 'momentum'
    return 'other'

cat_colors = {'volatility': '#e74c3c', 'trend': '#2ecc71', 'momentum': '#3498db', 'other': '#95a5a6'}
colors = [cat_colors[categorize(n)] for n in names_r]
ax.barh(range(top_n), aucs, color=colors)
ax.set_yticks(range(top_n))
ax.set_yticklabels(names_r, fontsize=7)
ax.axvline(x=0.5, color='gray', linestyle='--', alpha=0.5)
ax.set_xlabel('AUC (bust prediction)')
ax.set_title(f'All-Indicator Bust Prediction Ranking (top {top_n} of {len(ranked)})')

# Legend
from matplotlib.patches import Patch
legend_elements = [Patch(facecolor=c, label=k.title()) for k, c in cat_colors.items()]
ax.legend(handles=legend_elements, loc='lower right')
ax.grid(True, alpha=0.3)
plt.tight_layout()
savefig('27_auc_ranking.png')

# ══════════════════════════════════════════════════════════════
# 4. TEST TOP INDICATORS AS ENTRY GATES
# ══════════════════════════════════════════════════════════════
print("\n[4/5] Testing top 10 indicators as entry gates...")

top_10 = ranked[:10]
gate_results = {}

for name, info in top_10:
    vals = all_features[name]
    bust_higher = info['bust_higher']
    
    best_pf = 0
    best_result = None
    
    all_valid = vals[~np.isnan(vals)]
    if len(all_valid) < 100:
        continue
    
    for pct in [25, 30, 40, 50, 60, 70, 75, 80]:
        thresh = np.percentile(all_valid, pct)
        
        if bust_higher:
            filtered = [(idx, d) for idx, d in signals
                        if idx < len(vals) and (np.isnan(vals[idx]) or vals[idx] <= thresh)]
        else:
            filtered = [(idx, d) for idx, d in signals
                        if idx < len(vals) and (np.isnan(vals[idx]) or vals[idx] >= thresh)]
        
        if len(filtered) < 50:
            continue
        
        results = run_cycles_on_signals(candles, filtered, cfg)
        s = cycle_summary(results)
        
        if s['profit_factor'] > best_pf:
            best_pf = s['profit_factor']
            best_result = {
                **s,
                'threshold': round(float(thresh), 6),
                'percentile': pct,
                'pct_kept': round(len(filtered) / len(signals) * 100, 1),
                'bust_reduction': round((baseline['bust_rate'] - s['bust_rate']) / baseline['bust_rate'] * 100, 1)
                                  if baseline['bust_rate'] > 0 else 0,
            }
    
    if best_result:
        gate_results[name] = best_result
        print(f"    {name:35s}: bust↓{best_result['bust_reduction']:+.0f}%, "
              f"PF={best_result['profit_factor']:.2f}, kept={best_result['pct_kept']:.0f}%")

# Plot gates
fig, ax = plt.subplots(figsize=(12, 7))
gate_names = list(gate_results.keys())
reductions = [gate_results[n]['bust_reduction'] for n in gate_names]
pfs_g = [gate_results[n]['profit_factor'] for n in gate_names]

Phase 3 was 100% defensive: "chop hurts, avoid it."

This script flips the question:

  3. Quantify: how much equity comes from vol-expansion entries?
  4. Build a "vol confidence" score and test it as entry filter

The thesis: In volatile markets, price makes big moves in one direction.
TP triggers at L0 or L1 quickly. Fewer levels = less exposure = higher PF.
This is the OFFENSIVE complement to chop avoidance.

Outputs:

# ══════════════════════════════════════════════════════════════

def expansion_ratio(arr, lookback=20):
    """Current value / rolling mean. >1 = expanding."""
    out = np.full(len(arr), np.nan)
    for i in range(lookback, len(arr)):
        mean_val = np.nanmean(arr[i-lookback:i])
        if mean_val > 0 and not np.isnan(arr[i]):
            out[i] = arr[i] / mean_val
    return out

        'squeeze_mom': float(squeeze_mom[idx]) if idx < len(squeeze_mom) and not np.isnan(squeeze_mom[idx]) else np.nan,
        'adx_14': float(adx_14[idx]) if idx < len(adx_14) and not np.isnan(adx_14[idx]) else np.nan,

Build and test:
  1. "Confidence score" combining best bust predictor + best vol indicator
  2. Optimal entry: high confidence + vol expansion + EMA signal
  3. Full comparison: defensive-only vs offensive-only vs combined
  4. Validate on 2024-2025 out-of-sample

Outputs:
  - results/30_offensive_entry.json

  - plots/30_validation_2026.png

# ══════════════════════════════════════════════════════════════

    out = np.full(len(arr), np.nan)
    for i in range(lookback, len(arr)):
        mean_val = np.nanmean(arr[i-lookback:i])
        if mean_val > 0 and not np.isnan(arr[i]):
            out[i] = arr[i] / mean_val
    return out

#   - Trend strength (ADX > 20 → good, higher = better)
#   - Vol expansion (BBW expansion > 1 → good)
#   - Efficiency ratio (ER > threshold → trending, good)
#   - NOT choppy (chop_score < 0.5 → good)

def confidence_score(idx):
    """0-1 score: higher = better conditions for martingale entry."""
    if idx >= n:
        return np.nan
    
    score = 0.0
    count = 0
    
    # ADX component: 0-50 → 0-1
    if idx < len(adx_14) and not np.isnan(adx_14[idx]):
        score += np.clip(adx_14[idx] / 40, 0, 1)
        count += 1
    
    # BBW expansion: 0.5-2.0 → 0-1
    if idx < len(bbw_expansion) and not np.isnan(bbw_expansion[idx]):
        score += np.clip((bbw_expansion[idx] - 0.8) / 0.8, 0, 1)
        count += 1
    
    # ER: 0-0.3 → 0-1
    if idx < len(er_100) and not np.isnan(er_100[idx]):
        score += np.clip(er_100[idx] / 0.2, 0, 1)
        count += 1
    
    # Anti-chop: 0-1 → 1-0
    if idx < len(cs) and not np.isnan(cs[idx]):
        score += np.clip(1 - cs[idx] / 0.6, 0, 1)
        count += 1
    
    # ATR ratio: vol presence
    if idx < len(atr_ratio) and not np.isnan(atr_ratio[idx]):
        score += np.clip(atr_ratio[idx] / 1.2, 0, 1)
        count += 1
    
    return score / count if count > 0 else np.nan

    if idx >= tn:
        return np.nan
    score = 0.0
    count = 0
    if idx < len(test_adx) and not np.isnan(test_adx[idx]):
        score += np.clip(test_adx[idx] / 40, 0, 1)
        count += 1
    if idx < len(test_bbw_exp) and not np.isnan(test_bbw_exp[idx]):
        score += np.clip((test_bbw_exp[idx] - 0.8) / 0.8, 0, 1)
        count += 1
    if idx < len(test_er) and not np.isnan(test_er[idx]):
        score += np.clip(test_er[idx] / 0.2, 0, 1)
        count += 1
    if idx < len(test_cs) and not np.isnan(test_cs[idx]):
        score += np.clip(1 - test_cs[idx] / 0.6, 0, 1)
        count += 1
    if idx < len(test_atr_ratio) and not np.isnan(test_atr_ratio[idx]):
        score += np.clip(test_atr_ratio[idx] / 1.2, 0, 1)
        count += 1
    return score / count if count > 0 else np.nan
